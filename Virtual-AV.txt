# Install necessary packages
sudo pacman -S v4l2loopback-dkms alsa-utils ffmpeg --noconfirm

# Load Kernel modules
sudo modprobe v4l2loopback
sudo modprobe snd-aloop

# Unload Kernel modules
sudo modprobe -r v4l2loopback
sudo modprobe -r snd-aloop

---

# Creates a virtual video device
sudo modprobe v4l2loopback video_nr=2 card_label="ClientWebcam" exclusive_caps=1

# List custom devices
v4l2-ctl --list-devices

# Stream live feed to virtual video device
ffmpeg -re -i rtmp://*/* -f v4l2 -vcodec rawvideo -pix_fmt yuv420p /dev/video2
ffmpeg -re -i video.mp4 -f v4l2 -vcodec rawvideo -pix_fmt yuv420p /dev/video2

# Virtual video device passthrough via v4l2 in QEMU to guest
qemu-system-x86_64 ... -cameradev v4l2,path=/dev/video0,id=camera0
